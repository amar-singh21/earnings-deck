Programmatic KPI Check (Code Sample)

from decimal import Decimal

def gross_margin(revenue, cogs):
    revenue, cogs = Decimal(revenue), Decimal(cogs)
    return (revenue - cogs) / revenue

print(round(gross_margin('3200000000', '2320000000') * 100, 2))  # 27.5%
